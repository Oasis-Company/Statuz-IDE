#!/usr/bin/env node /** * Statuz IDE Legal Compliance Verification Script * Usage: node verification.cjs * Exit code 0 = all checks pass, non-zero = failures found */ 'use strict'; const fs = require('fs'); const path = require('path'); const { execSync } = require('child_process'); let failures = 0; const root = path.resolve(__dirname); function check(name, fn) { try { const result = fn(); if (result === true || result === 0) { console.log(' [PASS] ' + name); } else { console.log(' [FAIL] ' + name + ' - ' + (result || 'check failed')); failures++; } } catch (e) { console.log(' [FAIL] ' + name + ' - ' + e.message); failures++; } } function grep(pattern, filePattern) { try { const result = execSync( 'grep -rn "' + pattern + '" ' + filePattern + ' 2>nul', { cwd: root, encoding: 'utf8', maxBuffer: 10 * 1024 * 1024 } ); return result.trim() || ''; } catch (e) { return ''; } } function fileExists(f) { return fs.existsSync(path.join(root, f)); } console.log('\\n=== Statuz IDE Legal Compliance Verification ===\\n'); // 1. License Consistency console.log('--- License Consistency ---'); check('package.json license is Apache-2.0', () => { const pkg = JSON.parse(fs.readFileSync(path.join(root, 'package.json'), 'utf8')); return pkg.license === 'Apache-2.0'; }); check('product.json licenseName is Apache-2.0', () => { const prod = JSON.parse(fs.readFileSync(path.join(root, 'product.json'), 'utf8')); return prod.licenseName === 'Apache-2.0'; }); check('README.md contains Apache license description', () => { const readme = fs.readFileSync(path.join(root, 'README.md'), 'utf8'); return readme.includes('Apache License 2.0') || readme.includes('Apache-2.0'); }); // 2. No Glass Devtools Copyright console.log('\\n--- Glass Devtools Copyright ---'); check('No "Glass Devtools" in statuz module', () => { const r = grep('Glass Devtools', 'src/vs/workbench/contrib/statuz'); return r === '' ? true : 'Found: ' + r.substring(0, 200); }); // 3. No Void Trademark Residue console.log('\\n--- Void Trademark Residue ---'); check('No "Void Side Bar" in UI text', () => { const r = grep('Void Side Bar', 'src/'); return r === '' ? true : 'Found: ' + r.substring(0, 200); }); check('No "Void changed" comments', () => { const r = grep('Void changed', 'src/'); return r === '' ? true : 'Found: ' + r.substring(0, 200); }); check('No voideditor references in statuz module', () => { const r = grep('voideditor', 'src/vs/workbench/contrib/statuz'); return r === '' ? true : 'Found: ' + r.substring(0, 200); }); // 4. Copyright Headers console.log('\\n--- Copyright Headers ---'); check('All statuz .ts files have copyright', () => { const files = execSync( 'find src/vs/workbench/contrib/statuz/ -name "*.ts" -o -name "*.tsx" | head -200', { cwd: root, encoding: 'utf8' } ).trim().split('\\n').filter(Boolean); const missing = files.filter(f => { const content = fs.readFileSync(path.join(root, f), 'utf8'); return !content.includes('Copyright') && !content.includes('Apache'); }); return missing.length === 0 ? true : 'Missing: ' + missing.join(', '); }); // 5. Compliance Files console.log('\\n--- Compliance Files ---'); check('FFMPEG_LGPL_NOTICE.txt exists', () => fileExists('FFMPEG_LGPL_NOTICE.txt')); check('PRIVACY.md exists', () => fileExists('PRIVACY.md')); check('CONTRIBUTING.md exists', () => fileExists('CONTRIBUTING.md')); check('LEGAL.md exists', () => fileExists('LEGAL.md')); // 6. Extension Licenses console.log('\\n--- Extension Licenses ---'); check('open-remote-ssh has license', () => { const pkg = JSON.parse(fs.readFileSync( path.join(root, 'extensions/open-remote-ssh/package.json'), 'utf8' )); return !!pkg.license; }); check('open-remote-wsl has license', () => { const pkg = JSON.parse(fs.readFileSync( path.join(root, 'extensions/open-remote-wsl/package.json'), 'utf8' )); return !!pkg.license; }); // Summary console.log('\\n=== Verification Complete ==='); console.log('Failures: ' + failures); if (failures > 0) { console.log('SOME CHECKS FAILED. Review the output above.'); process.exit(1); } else { console.log('All checks passed!'); process.exit(0); }
运行方式：
node verification.cjs
